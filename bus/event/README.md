		return fmt.Errorf("–Ω–µ —É–¥–∞–ª–æ—Å—å –æ–±—Ä–∞–±–æ—Ç–∞—Ç—å –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è %s", e.UserID)
	}

}

// –û–±—Ä–∞–±–æ—Ç—á–∏–∫ –æ—à–∏–±–æ–∫ –¥–ª—è –∞—Å–∏–Ω—Ö—Ä–æ–Ω–Ω–æ–π –ø–æ–¥–ø–∏—Å–∫–∏.
errHandler := func(err error, e *UserCreatedEvent) {

}

// –ü–æ–¥–ø–∏—Å—ã–≤–∞–µ–º—Å—è —Å –æ–ø—Ü–∏—è–º–∏.
unsubscribe, err := userBus.Subscribe(
	asyncHandler,
	event.WithAsync[*UserCreatedEvent](),          // –í–∫–ª—é—á–∞–µ–º –∞—Å–∏–Ω—Ö—Ä–æ–Ω–Ω—ã–π —Ä–µ–∂–∏–º.
	event.WithErrorHandler(errHandler), // –ü—Ä–µ–¥–æ—Å—Ç–∞–≤–ª—è–µ–º –æ–±—Ä–∞–±–æ—Ç—á–∏–∫ –æ—à–∏–±–æ–∫.
)
if err != nil {
	panic(err)
}
defer unsubscribe()

// ... –ø—É–±–ª–∏–∫–∞—Ü–∏—è —Å–æ–±—ã—Ç–∏—è ...
